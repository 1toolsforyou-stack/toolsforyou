function getSecureRandom() { return window.crypto.getRandomValues(new Uint32Array(1))[0] / 4294967295; }

function generateSecureUUID() {
    if (typeof crypto !== 'undefined' && crypto.randomUUID) { try { return crypto.randomUUID(); } catch(e) {} }
    return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(c) {
        var r = getSecureRandom() * 16 | 0, v = c === 'x' ? r : (r & 0x3 | 0x8);
        return v.toString(16);
    });
}

const loadedScripts = new Set();
async function loadLibrary(url, timeout = 5000) {
    if (loadedScripts.has(url)) return Promise.resolve();
    return Promise.race([
        new Promise((resolve, reject) => {
            const script = document.createElement('script'); script.src = url;
            script.onload = () => { loadedScripts.add(url); resolve(); };
            script.onerror = reject; document.head.appendChild(script);
        }),
        new Promise((_, reject) => setTimeout(() => reject(new Error('Timeout loading ' + url)), timeout))
    ]).catch(err => {
        showToast("Error loading library. Check connection.");
        throw err;
    });
}

const dbPromise = new Promise((resolve, reject) => {

        const req = db.transaction('drafts').objectStore('drafts').get('mainInput');
        req.onsuccess = () => resolve(req.result || ""); req.onerror = () => resolve("");
    });
}

function handleHash() {
    const fullHash = window.location.hash.substring(1);
    const hashParts = fullHash.split('?');
    const hash = hashParts[0] || 'cleaner';
    const params = new URLSearchParams('?' + hashParts[1]);

    if (['about', 'contact', 'privacy', 'terms'].includes(hash)) switchView(hash);
    else if (hash.startsWith('article-')) loadArticle(hash);
    else if (registry[hash]) { switchView('home'); loadTool(hash, params.get('data')); }
    else { switchView('home'); loadTool('cleaner'); }
}

const wheelColors = ['#4f8cff', '#ff6b6b', '#4ade80', '#facc15', '#a855f7', '#fb923c', '#38bdf8', '#fb7185'];
let wheelRot = 0;
function drawWheel() {
    if(currentId !== 'random') return;
    const canvas = document.getElementById('wheelCanvas'); if(!canvas) return; const ctx = canvas.getContext('2d');
    let items = document.getElementById('mainInput').value.split('\n').filter(l => l.trim());
    if(!items.length) { ctx.clearRect(0,0,300,300); ctx.beginPath(); ctx.fillStyle='#e2e8f0'; ctx.arc(150,150,150,0,Math.PI*2); ctx.fill(); return; }
    const arc = Math.PI * 2 / items.length; ctx.clearRect(0, 0, 300, 300);
    for(let i=0; i<items.length; i++) {
        ctx.beginPath(); ctx.fillStyle = wheelColors[i % wheelColors.length]; ctx.moveTo(150, 150); ctx.arc(150, 150, 150, i * arc, (i + 1) * arc); ctx.lineTo(150, 150); ctx.fill();
        ctx.save(); ctx.translate(150, 150); ctx.rotate(i * arc + arc / 2); ctx.textAlign = "right"; ctx.fillStyle = "#fff"; ctx.font = "bold 14px 'Plus Jakarta Sans'"; ctx.fillText(items[i].substring(0, 15), 135, 5); ctx.restore();
    }
}
function spinWheel() {
    const items = document.getElementById('mainInput').value.split('\n').filter(l => l.trim());
    if(!items.length) return showToast("Enter at least 1 item!");
    const winIdx = Math.floor(getSecureRandom() * items.length); 
    const slice = 360 / items.length; const stopDeg = 360 - (winIdx * slice) - (slice / 2) + 270;
    wheelRot += 360 * 5; wheelRot = wheelRot - (wheelRot % 360) + stopDeg;
    const canvas = document.getElementById('wheelCanvas');
    canvas.style.transition = 'transform 3s cubic-bezier(0.17, 0.67, 0.12, 0.99)'; canvas.style.transform = `rotate(${wheelRot}deg)`;
    setTimeout(() => { displaySecure("🎉 Winner: " + items[winIdx]); }, 3000);
}
